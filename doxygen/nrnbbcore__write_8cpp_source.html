2 // A model built using NEURON is heavyweight in memory usage and that
3 // prevents maximizing the number of cells that can be simulated on
4 // a process. On the other hand, a tiny version of NEURON that contains
5 // only the cache efficient structures, minimal memory usage arrays,
6 // needed to do a simulation (no interpreter, hoc Objects, Section, etc.)
7 // lacks the model building flexibility of NEURON.
8 // Ideally, the only arrays needed for a tiny version simulation are those
9 // enumerated in the NrnThread structure in src/nrnoc/multicore.h up to,
10 // but not including, the Node** arrays. Also tiny versions of POINT_PROCESS,
11 // PreSyn, NetCon, and SUFFIX mechanisms will be stripped down from
12 // their full NEURON definitions and, it seems certain, many of the
13 // double fields will be converted to some other, less memory using, types.
14 // With this in mind, we envision that NEURON will incrementally construct
15 // cache efficient whole cell structures which can be saved and read with
16 // minimal processing into the tiny simulator. Note that this is a petabyte
17 // level of data volume. Consider, for example, 128K cores each
18 // preparing model data for several thousand cells using full NEURON where
19 // there is not enough space for the simultaneous existence of
20 // those several thousand cells --- but there is with the tiny version.

22 // Several assumptions with regard to the nrnbbcore_read reader.
23 // Since memory is filled with cells, whole cell
24 // load balance should be adequate and so there is no provision for
25 // multisplit. A process gets a list of the gid owned by that process
26 // and allocates the needed
27 // memory based on size variables for each gid, i.e.
28 // number of nodes, number of instances of each mechanism type, and number
29 // of NetCon instances. Also the offsets are calculated for where the
30 // cell information is to reside in the cache efficient arrays.
31 // The rest of the cell information is then copied
32 // into memory with the proper offsets. Pointers to data, used in the old
33 // NEURON world are converted to integer indices into a common data array.
34 
35 // A good deal of conceptual confusion resulted in earlier implementations
36 // with regard to ordering of synapses and
37 // artificial cells with and without gids. The ordering of the property
38 // data for those is defined by the order in the NrnThread.tml list where
39 // every Memb_list.data has an easily found index relative to its 'nodecount'.
40 // (For artificial cells, since those are not ordered in a cache efficient
41 // array, we get the index using int nrncore_art2index(double* param)
42 // which looks up the index in a hash table. Earlier implementations
43 // handled 'artificial cells without gids' specially which also
44 // necessitated special handling of their NetCons and disallowed artificial
45 // cells with gids. We now handle all artificial cells in a thread
46 // in the same way as any other synapse (the assumption still holds that
47 // any artificial cell without a gid in a thread can connect only to
48 // targets in the same thread. Thus, a single NrnThread.synapses now contains
49 // all synapses and all artificial cells belonging to that thread. All
50 // the synapses and artificial cells are in NrnThread.tml order. So there
51 // are no exceptions in filling Point_process pointers from the data indices
52 // on the coreneuron side. PreSyn ordering is a bit more delicate.
53 // From netpar.cpp, the gid2out_ hash table defines an output_gid
54 // ordering and gives us all the PreSyn
55 // associated with real and artificial cells having gids. But those are
56 // randomly ordered and interleaved with 'no gid instances'
57 // relative to the tml ordering.
58 // Since the number of output PreSyn >= number of output_gid it makes sense
59 // to order the PreSyn in the same way as defined by the tml ordering.
60 // Thus, even though artificial cells with and without gids are mixed,
61 // at least it is convenient to fill the PreSyn.psrc field.
62 // Synapses are first but the artificial cells with and without gids are
63 // mixed. The problem that needs to
64 // be explicitly overcome is associating output gids with the proper PreSyn
65 // and that can be done with a list parallel to the acell part of the
66 // output_gid list that specifies the PreSyn list indices.
67 // Note that allocation of large arrays allows considerable space savings
68 // by eliminating overhead involved in allocation of many individual
69 // instances.

755 // This function is related to stdindex2ptr in CoreNeuron to determine which values should

766  type = 0;

768  type = voltage; // signifies an index into voltage array portion of _data
769  index = pd - nt._actual_v;
770  } else if (pd >= nt._nrn_fast_imem->_nrn_sav_rhs && pd < (nt._nrn_fast_imem->_nrn_sav_rhs + nnode)) {
771  type = i_membrane_; // signifies an index into i_membrane_ array portion of _data
772  index = pd - nt._nrn_fast_imem->_nrn_sav_rhs;
773  }else{
774  for (NrnThreadMembList* tml = nt.tml; tml; tml = tml->next) {

776  Memb_list* ml1 = tml->ml;

778  if (pd >= ml1->data[0] && pd < (ml1->data[0] + nn)) {
779  type = tml->index;
780  index = pd - ml1->data[0];
781  break;
782  }
783  }
784  }

788 void datumindex_fill(int ith, CellGroup& cg, DatumIndices& di, Memb_list* ml) {
789  NrnThread& nt = nrn_threads[ith];
790  double* a = nt._actual_area;
791  int nnode = nt.end;
792  int mcnt = ml->nodecount;
793  int dsize = bbcore_dparam_size[di.type];
794  if (dsize == 0) { return; }
795  int* dmap = memb_func[di.type].dparam_semantics;
796  assert(dmap);
797  // what is the size of the nt._vdata portion needed for a single ml->dparam[i]
798  int vdata_size = 0;
799  for (int i=0; i < dsize; ++i) {
800  int* ds = memb_func[di.type].dparam_semantics;
801  if (ds[i] == -4 || ds[i] == -6 || ds[i] == -7 || ds[i] == 0) {
802  ++vdata_size;
803  }
804  }
805 
806  int isart = nrn_is_artificial_[di.type];
807  for (int i=0; i < mcnt; ++i) {
808  // Prop* datum instance arrays are not in cache efficient order
809  // ie. ml->pdata[i] are not laid out end to end in memory.
810  // Also, ml->data for artificial cells is not in cache efficient order
811  // but in the artcell case there are no pointers to doubles and
812  // the _actual_area pointer should be left unfilled.
813  Datum* dparam = ml->pdata[i];
814  int offset = i*dsize;
815  int vdata_offset = i*vdata_size;
816  for (int j=0; j < dsize; ++j) {
817  int etype = -100; // uninterpreted
818  int eindex = -1;
819  if (dmap[j] == -1) { // double* into _actual_area
820  if (isart) {
821  etype = -1;
822  eindex = -1; // the signal to ignore in bbcore.
823  }else{
824  if (dparam[j].pval == &ml->nodelist[i]->_area) {
825  // possibility it points directly into Node._area instead of
826  // _actual_area. For our purposes we need to figure out the
827  // _actual_area index.
828  etype = -1;
829  eindex = ml->nodeindices[i];
830  assert(a[ml->nodeindices[i]] == *dparam[j].pval);
831  }else{
832  if (dparam[j].pval < a || dparam[j].pval >= (a + nnode)){
833  printf("%s dparam=%p a=%p a+nnode=%p j=%d\n",
834  memb_func[di.type].sym->name, dparam[j].pval, a, a+nnode, j);
835  abort();
836  }
837  assert(dparam[j].pval >= a && dparam[j].pval < (a + nnode));
838  etype = -1;
839  eindex = dparam[j].pval - a;
840  }
841  }
842  }else if (dmap[j] == -2) { // this is an ion and dparam[j][0].i is the iontype
843  etype = -2;
844  eindex = dparam[j].i;
845  }else if (dmap[j] == -3) { // cvodeieq is always last and never seen
846  assert(dmap[j] != -3);
847  }else if (dmap[j] == -4) { // netsend (_tqitem pointer)
848  // eventually index into nt->_vdata
849  etype = -4;
850  eindex = vdata_offset++;
851  }else if (dmap[j] == -6) { // pntproc
852  // eventually index into nt->_vdata
853  etype = -6;
854  eindex = vdata_offset++;
855  }else if (dmap[j] == -7) { // bbcorepointer
856  // eventually index into nt->_vdata
857  etype = -6;
858  eindex = vdata_offset++;
859  }else if (dmap[j] == -8) { // watch
860  etype = -8;
861  eindex = 0;
862  }else if (dmap[j] == -9) { // diam
863  cg.ndiam = nt.end;
864  etype = -9;
865  // Rare for a mechanism to use dparam pointing to diam.
866  // MORPHOLOGY was never made cache efficient. And
867  // is not in the tml_with_art.
868  // Need to determine this node and then simple to search its
869  // mechanism list for MORPHOLOGY and then know the diam.
870  Node* nd = ml->nodelist[i];
871  double* pdiam = NULL;
872  for (Prop* p = nd->prop; p; p = p->next) {
873  if (p->type == MORPHOLOGY) {
874  pdiam = p->param;
875  break;
876  }
877  }
878  assert(dparam[j].pval == pdiam);
879  eindex = ml->nodeindices[i];
880  }else if (dmap[j] == -5) { // POINTER
881  // must be a pointer into nt->_data. Handling is similar to eion so
882  // give proper index into the type.
883  double* pd = dparam[j].pval;
884  nrn_dblpntr2nrncore(pd, nt, etype, eindex);
885  if (etype == 0) {
886  fprintf(stderr, "POINTER is not pointing to voltage or mechanism data. Perhaps it should be a BBCOREPOINTER\n");
887  }
888  assert(etype != 0);
889  // pointer into one of the tml types?
890  }else if (dmap[j] > 0 && dmap[j] < 1000) { // double* into eion type data
891  Memb_list* eml = cg.type2ml[dmap[j]];
892  assert(eml);
893  if(dparam[j].pval < eml->data[0]){
894  printf("%s dparam=%p data=%p j=%d etype=%d %s\n",
895  memb_func[di.type].sym->name, dparam[j].pval, eml->data[0], j,
896  dmap[j], memb_func[dmap[j]].sym->name);
897  abort();
898  }
899  assert(dparam[j].pval >= eml->data[0]);
900  etype = dmap[j];
901  if (dparam[j].pval >= (eml->data[0] +
902  (nrn_prop_param_size_[etype] * eml->nodecount))) {
903  printf("%s dparam=%p data=%p j=%d psize=%d nodecount=%d etype=%d %s\n",
904  memb_func[di.type].sym->name, dparam[j].pval, eml->data[0], j,
905  nrn_prop_param_size_[etype],
906  eml->nodecount, etype, memb_func[etype].sym->name);
907  }
908  assert(dparam[j].pval < (eml->data[0] +
909  (nrn_prop_param_size_[etype] * eml->nodecount)));
910  eindex = dparam[j].pval - eml->data[0];
911  }else if (dmap[j] > 1000) {//int* into ion dparam[xxx][0]
912  //store the actual ionstyle
913  etype = dmap[j];
914  eindex = *((int*)dparam[j]._pvoid);
915  } else {
916  char errmes[100];
917  sprintf(errmes, "Unknown semantics type %d for dparam item %d of", dmap[j], j);
918  hoc_execerror(errmes, memb_func[di.type].sym->name);
919  }
920  di.ion_type[offset + j] = etype;
921  di.ion_index[offset + j] = eindex;
922  }
923  }
